    <div class="row">

      <div>

  <div class="outcome" id="outcome" hidden>

    <p>Opening the status in a new tab. If it’s blocked, try one of these providers:</p>

    <div class="providers" id="providers"></div>

    <p class="small">Copy link fallback: <a class="inline" id="directUrl" href="#" target="_blank" rel="noopener"></a></p>

  <script>

    // Prefill today

    const dateInput = document.getElementById('date');

    if (!dateInput.value) {

      const now = new Date();

      const yyyy = now.getFullYear();

      const mm = String(now.getMonth() + 1).padStart(2, '0');

      const dd = String(now.getDate()).padStart(2, '0');

      dateInput.value = `${yyyy}-${mm}-${dd}`;

    }


    const resetBtn = document.getElementById('resetBtn');

    resetBtn.addEventListener('click', () => {

      document.getElementById('flightForm').reset();

      const now = new Date();

      const yyyy = now.getFullYear();

      const mm = String(now.getMonth() + 1).padStart(2, '0');

      const dd = String(now.getDate()).padStart(2, '0');

      dateInput.value = `${yyyy}-${mm}-${dd}`;

      document.getElementById('outcome').hidden = true;

      document.getElementById('providers').innerHTML = '';

    });


    const openNewTab = (url) => {

      // Open in new tab to bypass iframe/X-Frame-Options

      window.open(url, '_blank', 'noopener'); 

    };


    const mkBtn = (label, url) => {

      const b = document.createElement('button');

      b.type = 'button';

      b.className = 'btn link';

      b.textContent = label;

      b.addEventListener('click', () => openNewTab(url));

      return b;

    };


    document.getElementById('flightForm').addEventListener('submit', (e) => {

      e.preventDefault();


      const airline = (document.getElementById('airline').value || 'AF').trim().toUpperCase();

      const flightNoRaw = (document.getElementById('flightNo').value || '').trim().toUpperCase();

      const dep = (document.getElementById('dep').value || '').trim().toUpperCase();

      const arr = (document.getElementById('arr').value || '').trim().toUpperCase();

      const dateStr = document.getElementById('date').value;


      if (!dateStr) { alert('Please select a date'); return; }

      const [year, monthStr, dayStr] = dateStr.split('-');

      const yearNum = parseInt(year, 10);

      const monthNum = parseInt(monthStr, 10);

      const dayNum = parseInt(dayStr, 10);


      let airlineCode = airline;

      let flightNumberDigits = '';

      if (flightNoRaw) {

        const m = flightNoRaw.match(/^([A-Z]{1,3})?\s*0*([0-9]{1,4})$/);

        if (!m) { alert('Invalid flight number. Examples: AF83 or 83'); return; }

        if (m[1]) airlineCode = m[1].toUpperCase();

        flightNumberDigits = m[2];

      }


      let urlFlightStats = '';

      let urlAF = '';

      let urlFlightAware = '';

      let urlFR24 = '';


      if (flightNumberDigits) {

        urlFlightStats = `https://www.flightstats.com/v2/flight-tracker/${encodeURIComponent(airlineCode)}/${encodeURIComponent(flightNumberDigits)}?year=${yearNum}&month=${monthNum}&date=${dayNum}`;

        // Alternatives by flight number

        const icao = airlineCode === 'AF' ? 'AFR' : airlineCode; // common mapping, AF -> AFR on many trackers

        urlAF = `https://wwws.airfrance.fr/en/flight-status`; // no stable prefill, open landing

        urlFlightAware = `https://flightaware.com/live/flight/${encodeURIComponent(icao)}${encodeURIComponent(flightNumberDigits)}`;

        urlFR24 = `https://www.flightradar24.com/data/flights/${encodeURIComponent(airlineCode.toLowerCase())}${encodeURIComponent(flightNumberDigits)}`;

      } else {

        if (!dep || !arr || dep.length !== 3 || arr.length !== 3) {

          alert('Please enter valid IATA codes for departure and arrival (3 letters each, e.g. CDG and JFK)');

          return;

        }

        urlFlightStats = `https://www.flightstats.com/v2/flight-tracker/route/${encodeURIComponent(dep)}/${encodeURIComponent(arr)}?year=${yearNum}&month=${monthNum}&date=${dayNum}`;

        // Alternatives by route

        urlAF = `https://wwws.airfrance.fr/en/flight-status`;

        urlFlightAware = `https://flightaware.com/live/findflight?origin=${encodeURIComponent(dep)}&destination=${encodeURIComponent(arr)}`;

        urlFR24 = `https://www.flightradar24.com/route/${encodeURIComponent(dep)}/${encodeURIComponent(arr)}`;

      }


      // Open primary target in new tab

      openNewTab(urlFlightStats);


      // Show alternatives on screen to click if needed

      const outcome = document.getElementById('outcome');

      outcome.hidden = false;

      document.getElementById('directUrl').textContent = urlFlightStats;

      document.getElementById('directUrl').href = urlFlightStats;


      const providers = document.getElementById('providers');

      providers.innerHTML = '';

      providers.appendChild(mkBtn('Air France', urlAF));

      providers.appendChild(mkBtn('FlightAware', urlFlightAware));

      providers.appendChild(mkBtn('Flightradar24', urlFR24));

      providers.appendChild(mkBtn('FlightStats', urlFlightStats));

    });

  </script>
